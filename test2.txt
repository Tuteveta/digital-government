function readFile(p) { return fs.readFileSync(p, 'utf8'); }

function buildBreadcrumbs(parents) {
  let out = '';
  for (const p of parents) {
    out += '
              <Link href="' + '{p.href_PLACEHOLDER}' + '" className="hover:text-white transition-colors">' + '{p.label_PLACEHOLDER}' + '</Link>
              <ChevronRight className="w-4 h-4" />';
  }
  return out;
}
